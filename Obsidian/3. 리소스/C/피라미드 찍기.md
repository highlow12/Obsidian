[[02. 반복문]] 을 사용하여 해결한다

# code
```c
#include <stdio.h>

int main()
{
    // 입력부
    int floor = 0;
    printf("층 수를 입력하세요: ");
    scanf_s("%d", &floor);
    // 출력부

    for (int i = 1; i <= floor ; i++)
    {
        // 공백 출력
        for (int j = floor - i; j > 0; j--)
        {
            printf(" ");
        }
  
        //*출력
        for (int k = 0; k < 1 + (i-1) * 2; k++)
        {
            printf("*");
        }
        printf("\n");
    }
    return 0;
}
```
