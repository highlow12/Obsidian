function insertAtEnd(head, element):
    newNode = new Node
    newNode.element = element
    newNode.next = null
    if isEmpty(head):
        head = newNode
    else:
        current = head
        while current.next != null:
            current = current.next
        current.next = newNode
    return head

function findElement(head, element):
    position = 0
    current = head
    while current != null:
        if current.element == element:
            return position
        current = current.next
        position += 1
    return -1  // 원소를 찾지 못했을 때
